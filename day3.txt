version: "3.8"

services:
  mysql:
    image: mysql:5.6
    container_name: mysql-standalone
    env_file:
      - mysql.env
    ports:
      - "3306:3306"
    volumes:
      - myvol:/var/lib/mysql

  app:
    build: .
    container_name: users-mysql-app
    env_file:
      - mysql.env
    environment:
      SPRING_DATASOURCE_URL: jdbc:mysql://mysql:3306/${MYSQL_DATABASE}
      SPRING_DATASOURCE_USERNAME: ${MYSQL_USER}
      SPRING_DATASOURCE_PASSWORD: ${MYSQL_PASSWORD}
    ports:
      - "8086:8086"
    depends_on:
      - mysql

volumes:
  myvol:
    external: true
